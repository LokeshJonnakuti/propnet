import numpy as np



[docs]def plug_in(symbol_values):

    req_symbols = ["S", "e", "d"]
    data = {}
    if all(s in symbol_values for s in req_symbols):
        e = symbol_values["e"]
        S = symbol_values["S"]
        d = symbol_values["d"]

        data["k"] = np.abs(d[2][2] / np.sqrt(e[2][2] * S[2][2]))

    return data
